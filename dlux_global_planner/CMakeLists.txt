cmake_minimum_required(VERSION 2.8.3)
project(dlux_global_planner)
set_directory_properties(PROPERTIES COMPILE_OPTIONS "-std=c++11;-Wall;-Werror")
find_package(catkin REQUIRED COMPONENTS
        geometry_msgs
        nav_2d_msgs
        nav_2d_utils
        nav_core2
        nav_grid
        nav_grid_pub_sub
        nav_msgs
        pluginlib
        roscpp
        visualization_msgs
)
catkin_package(
    CATKIN_DEPENDS
        geometry_msgs
        nav_2d_msgs
        nav_2d_utils
        nav_core2
        nav_grid
        nav_grid_pub_sub
        nav_msgs
        pluginlib
        roscpp
        visualization_msgs
    INCLUDE_DIRS
        include
    LIBRARIES
        dgp
)

add_library(dgp src/dlux_global_planner.cpp src/cost_interpreter.cpp)
add_dependencies(dgp ${catkin_EXPORTED_TARGETS})
target_link_libraries(dgp ${catkin_LIBRARIES})

add_executable(planner_node src/planner_node.cpp)
add_dependencies(planner_node ${catkin_EXPORTED_TARGETS})
target_link_libraries(planner_node dgp ${catkin_LIBRARIES})

include_directories(include ${catkin_INCLUDE_DIRS})

if (CATKIN_ENABLE_TESTING)
  find_package(roslint)
  roslint_cpp()
  roslint_add_test()

  find_package(rostest REQUIRED)
  catkin_add_gtest(kernel_test test/kernel_test.cpp)
  target_link_libraries(kernel_test dgp)

  find_package(global_planner_tests)
  include_directories(include ${global_planner_tests_INCLUDE_DIRS} ${catkin_INCLUDE_DIRS} ${GTEST_INCLUDE_DIRS})
  link_directories(${GTEST_LIBRARY_DIRS})

  add_rostest_gtest(got test/global_oscillation_test.launch test/global_oscillation_test.cpp)
  target_link_libraries(got dgp ${global_planner_tests_LIBRARIES} ${catkin_LIBRARIES} ${GTEST_LIBRARIES})
endif()

install(TARGETS planner_node
        RUNTIME DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)
install(TARGETS dgp
        ARCHIVE DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
        LIBRARY DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
        RUNTIME DESTINATION ${CATKIN_GLOBAL_BIN_DESTINATION}
)
install(DIRECTORY include/${PROJECT_NAME}/
        DESTINATION ${CATKIN_PACKAGE_INCLUDE_DESTINATION}
)
install(FILES nav_core2_plugins.xml
        DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION}
)
